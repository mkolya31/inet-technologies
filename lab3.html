Настройка маршрутизаторов

Настройка Router1:

enable
configure terminal
interface fa0/1
ip address 192.168.0.1 255.255.255.128
no shutdown
exit
interface fa1/0
ip address 172.16.0.1 255.255.255.240
no shutdown
exit
interface fa0/0
ip address 15.15.0.1 255.255.255.252
no shutdown
exit
ip route 10.10.0.0 255.255.252.0 fa0/0
enable password router1
line console 0
password router1
login
exit
line vty 0 15
password router1
login
exit
service password-encryption
banner motd "Welcome to Router1"

Настройка Router2:

enable
configure terminal
interface fa0/1
ip address 10.10.0.1 255.255.252.0
no shutdown
exit
interface fa0/0
ip address 15.15.0.2 255.255.255.252
no shutdown
exit
ip route 192.168.0.0 255.255.255.128 fa0/0
ip route 172.16.0.0 255.255.255.240 fa0/0
enable password router2
line console 0
password router2
login
exit
line vty 0 15
password router2
login
exit
service password-encryption
banner motd "Welcome to Router2"
